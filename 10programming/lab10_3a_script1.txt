-- Exercise 10.3.a

SQL> lock table Movie IN ROW SHARE MODE;

Table(s) Locked.


SQL> select rank from Movie where id = 238072;

      RANK
----------
         9

-- updating the table within the lock
SQL> update Movie set rank=10 where id = 238072;

1 row updated.

SQL> select rank from Movie where id = 238072;

      RANK
----------
        10
-- unlocking the table
SQL> commit;

Commit complete.
